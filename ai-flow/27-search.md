		return false

		case "gt":
			if !(n.SizeBytes > val) {
				return false
			}
		case "gte":
			if !(n.SizeBytes >= val) {
				return false
			}
		case "lt":
			if !(n.SizeBytes < val) {

			}
		case "lte":
			if !(n.SizeBytes <= val) {

			}
		case "eq":
			if n.SizeBytes != val {

			}
		}
	}

	// created/modified Ğ´Ğ¸Ğ°Ğ¿Ğ°Ğ·Ğ¾Ğ½Ñ‹ (ĞµÑĞ»Ğ¸ Ğ·Ğ°Ğ´Ğ°Ğ½Ñ‹)

			}
		case "lt":
			if !n.Created.Before(t) {
				return false
			}
		}
	}

		switch op {
		case "gt":
			if !n.Updated.After(t) {
				return false
			}
		case "lt":
			if !n.Updated.Before(t) {

			}
		}
	}

func parseSizeFilters(q url.Values) map[string]int64 {
    m := make(map[string]int64)

    return n
